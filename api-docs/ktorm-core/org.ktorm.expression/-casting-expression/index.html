CastingExpression

data class CastingExpression<T : Any>(val expression: SqlExpression, val sqlType: SqlType<T>, val isLeafNode: Boolean = false, val extraProperties: Map<String, Any> = emptyMap()) : ScalarExpression<T> (source)
